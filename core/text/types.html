\[\begin{split}\begin{array}{llclll} \def\mathdef4028#1{{}}\mathdef4028{recursive type} & \href{../text/types.html#text-typedef}{\mathtt{typedef}} &\equiv& \def\mathdef4220#1{\mbox{‘}\mathtt{#1}\mbox{’}}\mathdef4220{(}~~\def\mathdef4221#1{\mbox{‘}\mathtt{#1}\mbox{’}}\mathdef4221{rec}~~\href{../text/types.html#text-typedef}{\mathtt{typedef}}~~\def\mathdef4222#1{\mbox{‘}\mathtt{#1}\mbox{’}}\mathdef4222{)} \\ \end{array}\end{split}\]

\[\begin{split}\begin{array}{llclll} \def\mathdef4028#1{{}}\mathdef4028{sub type} & \href{../text/types.html#text-comptype}{\mathtt{comptype}} &\equiv& \def\mathdef4223#1{\mbox{‘}\mathtt{#1}\mbox{’}}\mathdef4223{(}~~\def\mathdef4224#1{\mbox{‘}\mathtt{#1}\mbox{’}}\mathdef4224{sub}~~\def\mathdef4225#1{\mbox{‘}\mathtt{#1}\mbox{’}}\mathdef4225{final}~~\epsilon~~\href{../text/types.html#text-comptype}{\mathtt{comptype}}~~\def\mathdef4226#1{\mbox{‘}\mathtt{#1}\mbox{’}}\mathdef4226{)} \\ \end{array}\end{split}\]

Abbreviations¶

\[\begin{array}{llclll} \def\mathdef4028#1{{}}\mathdef4028{address type} & \def\mathdef4229#1{\mbox{‘}\mathtt{#1}\mbox{’}}\mathdef4229{} &\equiv& \def\mathdef4230#1{\mbox{‘}\mathtt{#1}\mbox{’}}\mathdef4230{i32} \end{array}\]
