\[\begin{split}\begin{array}{llclll} \def\mathdef4043#1{{}}\mathdef4043{recursive type} & \href{../text/types.html#text-typedef}{\mathtt{typedef}} &\equiv& \def\mathdef4235#1{\mbox{‘}\mathtt{#1}\mbox{’}}\mathdef4235{(}~~\def\mathdef4236#1{\mbox{‘}\mathtt{#1}\mbox{’}}\mathdef4236{rec}~~\href{../text/types.html#text-typedef}{\mathtt{typedef}}~~\def\mathdef4237#1{\mbox{‘}\mathtt{#1}\mbox{’}}\mathdef4237{)} \\ \end{array}\end{split}\]

\[\begin{split}\begin{array}{llclll} \def\mathdef4043#1{{}}\mathdef4043{sub type} & \href{../text/types.html#text-comptype}{\mathtt{comptype}} &\equiv& \def\mathdef4238#1{\mbox{‘}\mathtt{#1}\mbox{’}}\mathdef4238{(}~~\def\mathdef4239#1{\mbox{‘}\mathtt{#1}\mbox{’}}\mathdef4239{sub}~~\def\mathdef4240#1{\mbox{‘}\mathtt{#1}\mbox{’}}\mathdef4240{final}~~\epsilon~~\href{../text/types.html#text-comptype}{\mathtt{comptype}}~~\def\mathdef4241#1{\mbox{‘}\mathtt{#1}\mbox{’}}\mathdef4241{)} \\ \end{array}\end{split}\]
